Generar planeación de ejecución de etapa│├── Obtener datos│    ├── Captura de cantidades (carpetas)│    ├── Captura de personal (analistas y supervisores)│    ├── Captura de fechas (inicio y final)│    ├── Captura de metas por rol│    └── Validar datos de entrada│├── Calcular variables base│    ├── Calcular meta diaria grupal por rol│    └── Calcular días calendario│├── Calcular proyección diaria (por cada día)│    ├── Calcular fecha del día│    ├── Calcular meta individual analista│    ├── Calcular meta individual supervisor│    ├── Calcular meta grupal analista│    ├── Calcular meta grupal supervisor│    └── Calcular % avance│└── Mostrar resultados     ├── Mostrar Tabla 0 — descripción de la etapa     ├── Mostrar Tabla 1 — planeación por día     └── Mostrar Tabla 2 — proyección de la etapa______________________________________________________________________________________________________________________________________________________________________________________________________________________________________________from rich.table import Tablefrom rich.console import Console# ── ENTRADA ──COL                  = 20etapa                = "Valoracion de Antecedentes"cantidad_carpetas    = 5015cantidad_analistas   = 20cantidad_supervisores = 10fecha_inicio         = 1fecha_final          = 23meta_diaria_analistas    = 12meta_diaria_supervisores = 24# ── VARIABLES BASE ──meta_diaria_grupal_analistas    = cantidad_analistas    * meta_diaria_analistasmeta_diaria_grupal_supervisores = cantidad_supervisores * meta_diaria_supervisoresdias = fecha_final - fecha_inicio# ── TABLA 0 — Descripción de la etapa ──t0 = Table(title="DESCRIPCION DE LA ETAPA", show_header=False)t0.add_column("Campo",  style="bold")t0.add_column("Valor")t0.add_row("Etapa",                etapa)t0.add_row("Aspirantes proyectados", f"{cantidad_carpetas:,}")t0.add_row("Días calendario",      f"{dias}")Console().print(t0)# ── TABLA 1 — Planeación por día ──t1 = Table(title="\nPLANEACION POR DIA")t1.add_column("ROL",                  style="bold")t1.add_column("Personas",             justify="center")t1.add_column("Meta Individual/Día",  justify="center")t1.add_column("Meta Grupal/Día",      justify="center")t1.add_row("ANALISTA",    f"{cantidad_analistas}",    f"{meta_diaria_analistas}",    f"{meta_diaria_grupal_analistas:,}")t1.add_row("SUPERVISOR",  f"{cantidad_supervisores}", f"{meta_diaria_supervisores}", f"{meta_diaria_grupal_supervisores:,}")Console().print(t1)# ── TABLA 2 — Proyección incremental ──t2 = Table(title="\nPLANEACION PARA LA EJECUCION DE LA ETAPA")t2.add_column("Fecha",           justify="center")t2.add_column("Meta Ind. A",     justify="center")t2.add_column("Meta Ind. S",     justify="center")t2.add_column("Meta Grupal A",   justify="center")t2.add_column("Meta Grupal S",   justify="center")t2.add_column("% Avance",        justify="center")for i in range(0, fecha_final - fecha_inicio):    fecha_del_dia = fecha_inicio + i    # Analista con límite    meta_individual_analista = meta_diaria_analistas * (i + 1)    meta_grupal_analistas    = meta_individual_analista * cantidad_analistas    if meta_grupal_analistas >= cantidad_carpetas:        meta_grupal_analistas    = cantidad_carpetas        meta_individual_analista = meta_grupal_analistas / cantidad_analistas    # Supervisor con gabela (i=0 produce 0 automáticamente)    meta_individual_supervisor  = meta_diaria_supervisores * i    meta_grupal_supervisores    = meta_individual_supervisor * cantidad_supervisores    if meta_grupal_supervisores >= cantidad_carpetas:        meta_grupal_supervisores    = cantidad_carpetas        meta_individual_supervisor  = meta_grupal_supervisores / cantidad_supervisores    avance = meta_grupal_supervisores / cantidad_carpetas    t2.add_row(        f"{fecha_del_dia}",        f"{meta_individual_analista:,.0f}",        f"{meta_individual_supervisor:,.0f}",        f"{meta_grupal_analistas:,.0f}",        f"{meta_grupal_supervisores:,.0f}",        f"{avance:.2%}"    )Console().print(t2)
