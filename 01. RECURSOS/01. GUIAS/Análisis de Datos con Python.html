2.1 — Trabajar con Distribuciones scipy.stats

02_distribuciones.py python 
from scipy import stats
import numpy as np
import matplotlib.pyplot as plt

# ════════════════════════════════════════════════════════
# DISTRIBUCIÓN NORMAL
# ════════════════════════════════════════════════════════
mu, sigma = 850_000, 150_000  # media y desv. std del precio
dist_normal = stats.norm(loc=mu, scale=sigma)

# Probabilidad de que el precio sea menor a $700,000
p_menos_700k = dist_normal.cdf(700_000)
print(f"P(precio < $700k) = {p_menos_700k:.4f} ({p_menos_700k*100:.1f}%)")

# Probabilidad entre $700k y $1,000k (68-95-99.7 verificado)
p_rango = dist_normal.cdf(1_000_000) - dist_normal.cdf(700_000)
print(f"P($700k < precio < $1M) = {p_rango:.4f}")

# Valor que deja el 95% abajo (percentil 95)
p95 = dist_normal.ppf(0.95)
print(f"Percentil 95: ${p95:,.0f}")

# Generar muestra aleatoria de precios
precios_sim = dist_normal.rvs(size=1000, random_state=42)

# ════════════════════════════════════════════════════════
# DISTRIBUCIÓN BINOMIAL
# ════════════════════════════════════════════════════════
# Si el 30% de ventas tienen descuento, en 20 ventas,
# ¿cuál es la prob. de que exactamente 5 tengan descuento?
dist_bin = stats.binom(n=20, p=0.30)
p_exacta = dist_bin.pmf(5)
p_menos5 = dist_bin.cdf(5)
print(f"\nBinomial — P(X=5) = {p_exacta:.4f}")
print(f"Binomial — P(X≤5) = {p_menos5:.4f}")

# ════════════════════════════════════════════════════════
# DISTRIBUCIÓN POISSON
# ════════════════════════════════════════════════════════
# Promedio de 8 pedidos por día, P(X = 12 pedidos)
dist_pois = stats.poisson(mu=8)
print(f"\nPoisson  — P(X=12) = {dist_pois.pmf(12):.4f}")
print(f"Poisson  — P(X>10) = {1 - dist_pois.cdf(10):.4f}")

2.2 — Intervalos de Confianza scipy.stats

02_intervalos_confianza.py python 
from scipy import stats
import numpy as np

muestra = df["total_cop"]
n       = len(muestra)
x_bar   = muestra.mean()
s       = muestra.std(ddof=1)
alpha   = 0.05  # nivel de significancia (IC del 95%)

# ── IC con distribución t (σ desconocida) ────────────────
ic_inferior, ic_superior = stats.t.interval(
    confidence = 1 - alpha,
    df         = n - 1,        # grados de libertad
    loc        = x_bar,
    scale      = s / np.sqrt(n)  # error estándar
)

print(f"Media muestral   : ${x_bar:,.0f}")
print(f"IC 95% inferior  : ${ic_inferior:,.0f}")
print(f"IC 95% superior  : ${ic_superior:,.0f}")
print(f"Margen de error  : ${(ic_superior - x_bar):,.0f}")

# ── IC para diferentes niveles de confianza ──────────────
for conf in [0.90, 0.95, 0.99]:
    lo, hi = stats.t.interval(conf, df=n-1, loc=x_bar, scale=s/np.sqrt(n))
    print(f"IC {conf*100:.0f}%: [${lo:,.0f} — ${hi:,.0f}]")
▶ output Media muestral : $3,987,500 IC 95% inferior : $3,681,034 IC 95% superior : $4,293,966 Margen de error : $306,466 IC 90%: [$3,731,280 — $4,243,720] IC 95%: [$3,681,034 — $4,293,966] IC 99%: [$3,580,142 — $4,394,858]

2.3 — Pruebas de Hipótesis scipy.stats

02_hipotesis.py python 
from scipy import stats

alpha = 0.05

# ════════════════════════════════════════════════════════
# PRUEBA 1: t de una muestra
# H₀: La venta promedio es $4,000,000
# H₁: La venta promedio ES DIFERENTE de $4,000,000
# ════════════════════════════════════════════════════════
t_stat, p_val = stats.ttest_1samp(df["total_cop"], popmean=4_000_000)
print("── Prueba t (una muestra) ──────────────────")
print(f"t = {t_stat:.4f}  |  p-valor = {p_val:.4f}")
print("Conclusión:", "Rechazar H₀" if p_val < alpha else "No rechazar H₀")

# ════════════════════════════════════════════════════════
# PRUEBA 2: t de dos muestras independientes
# H₀: Ana y Luis tienen el mismo promedio de ventas
# H₁: Sus promedios son diferentes
# ════════════════════════════════════════════════════════
ventas_ana  = df[df["vendedor"] == "Ana"]["total_cop"]
ventas_luis = df[df["vendedor"] == "Luis"]["total_cop"]

t_stat2, p_val2 = stats.ttest_ind(ventas_ana, ventas_luis, equal_var=False)
print("\n── Prueba t (dos muestras) ─────────────────")
print(f"t = {t_stat2:.4f}  |  p-valor = {p_val2:.4f}")
print("Conclusión:", "Rechazar H₀" if p_val2 < alpha else "No rechazar H₀")

# ════════════════════════════════════════════════════════
# PRUEBA 3: ANOVA — ¿Los 4 vendedores tienen el mismo promedio?
# H₀: μ_Ana = μ_Luis = μ_Carlos = μ_María
# H₁: Al menos un vendedor tiene promedio diferente
# ════════════════════════════════════════════════════════
grupos = [df[df["vendedor"]==v]["total_cop"] for v in df["vendedor"].unique()]
f_stat, p_anova = stats.f_oneway(*grupos)
print("\n── ANOVA (4 vendedores) ────────────────────")
print(f"F = {f_stat:.4f}  |  p-valor = {p_anova:.4f}")
print("Conclusión:", "Diferencia significativa" if p_anova < alpha else "Sin diferencia significativa")

# ════════════════════════════════════════════════════════
# PRUEBA 4: Chi-cuadrado — ¿Producto y descuento son independientes?
# ════════════════════════════════════════════════════════
tabla_obs = pd.crosstab(df["producto"], df["descuento"])
chi2, p_chi, gl, esperados = stats.chi2_contingency(tabla_obs)
print("\n── Chi-cuadrado (independencia) ────────────")
print(f"χ² = {chi2:.4f}  |  p-valor = {p_chi:.4f}  |  gl = {gl}")
print("Conclusión:", "Dependientes" if p_chi < alpha else "Independientes")
▶ output ── Prueba t (una muestra) ────────────────── t = -0.0891 | p-valor = 0.9291 Conclusión: No rechazar H₀ # El promedio NO difiere significativamente de $4,000,000 ── Prueba t (dos muestras) ───────────────── t = -0.4012 | p-valor = 0.6898 Conclusión: No rechazar H₀ # No hay evidencia de que Ana y Luis sean diferentes ── ANOVA (4 vendedores) ──────────────────── F = 0.2134 | p-valor = 0.8872 Conclusión: Sin diferencia significativa ── Chi-cuadrado (independencia) ──────────── χ² = 3.1420 | p-valor = 0.7911 | gl = 9 Conclusión: Independientes

import numpy as np

# ── Crear vectores ───────────────────────────────────────
u = np.array([3, -1, 5])
v = np.array([2,  4, -2])

# ── Operaciones ──────────────────────────────────────────
suma        = u + v                    # [5, 3, 3]
escalar     = 3 * u                    # [9, -3, 15]
dot_product = np.dot(u, v)            # u·v = 3*2 + (-1)*4 + 5*(-2) = -8
norma_u     = np.linalg.norm(u)      # ||u|| = sqrt(9+1+25) = 5.916

# ── Ángulo entre vectores ────────────────────────────────
cos_theta = dot_product / (np.linalg.norm(u) * np.linalg.norm(v))
angulo_deg = np.degrees(np.arccos(np.clip(cos_theta, -1, 1)))

print(f"u + v         = {suma}")
print(f"Producto punto= {dot_product}")
print(f"||u||         = {norma_u:.4f}")
print(f"Ángulo u,v    = {angulo_deg:.2f}°")
print(f"Ortogonales?  {np.isclose(dot_product, 0)}")

3.3 — Sistemas de Ecuaciones Lineales numpy.linalg

03_sistemas.py python 
import numpy as np

# Sistema:  2x + y  = 8
#            x + 3y = 10
#
# Forma matricial: Ax = b

A = np.array([[2, 1],
              [1, 3]])
b = np.array([8, 10])

# ── Método 1: np.linalg.solve (más estable numéricamente)
x = np.linalg.solve(A, b)
print(f"Solución: x={x[0]:.4f}, y={x[1]:.4f}")
print(f"Verificación A@x = {A @ x}")

# ── Método 2: Usando la inversa (x = A⁻¹b) ──────────────
x_inv = np.linalg.inv(A) @ b
print(f"Por inversa: x={x_inv[0]:.4f}, y={x_inv[1]:.4f}")

# ── Caso práctico: precios de 3 productos ────────────────
# Semana 1: 2 Laptops + 1 Monitor = $3,200,000
# Semana 2: 1 Laptop  + 2 Monitores = $2,600,000
# ¿Cuál es el precio de cada uno?
A_precios = np.array([[2, 1],

3.4 — Valores y Vectores Propios numpy · sklearn PCA

03_eigenvalues_pca.py python 
import numpy as np
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler

# ── Eigenvalues de la matriz de covarianza ────────────────
numericas = df[["cantidad", "precio", "descuento", "total_cop"]].values

# Estandarizar (media=0, std=1) — requisito para PCA
scaler    = StandardScaler()
X_scaled  = scaler.fit_transform(numericas)

# Calcular covarianza y sus eigenvalues manualmente
cov_matrix = np.cov(X_scaled.T)
eigenvalues, eigenvectors = np.linalg.eig(cov_matrix)

print("Eigenvalues (varianza explicada por cada componente):")
for i, val in enumerate(eigenvalues):
    print(f"  λ{i+1} = {val:.4f}  ({val/sum(eigenvalues)*100:.1f}%)")

# ── PCA con scikit-learn (mucho más sencillo) ─────────────
pca = PCA(n_components=2)   # reducir a 2 dimensiones
X_pca = pca.fit_transform(X_scaled)

print("\nPCA scikit-learn:")
print(f"Varianza explicada: {pca.explained_variance_ratio_*100}")
print(f"Varianza acumulada: {pca.explained_variance_ratio_.sum()*100:.1f}%")
print(f"Forma original: {X_scaled.shape} → Reducida: {X_pca.shape}")

# Visualizar las 2 primeras componentes principales
import matplotlib.pyplot as plt
plt.figure(figsize=(8, 5))
plt.scatter(X_pca[:,0], X_pca[:,1], alpha=.5, s=20)
plt.xlabel(f"PC1 ({pca.explained_variance_ratio_[0]*100:.1f}%)")
plt.ylabel(f"PC2 ({pca.explained_variance_ratio_[1]*100:.1f}%)")

4.1 — Regresión Lineal: La Fórmula Matricial en Acción sklearn

La regresión lineal múltiple usa exactamente la fórmula del Módulo III: β = (XᵀX)⁻¹Xᵀy. scikit-learn la resuelve numéricamente de forma estable.

04_regresion_lineal.py python 
import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

# ── 1. Preparar features y target ───────────────────────
X = df[["cantidad", "precio", "descuento"]]
y = df["total_cop"]

# ── 2. Dividir en entrenamiento (80%) y prueba (20%) ─────
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.20, random_state=42
)

# ── 3. Estandarizar features ─────────────────────────────
scaler  = StandardScaler()
X_train_s = scaler.fit_transform(X_train)
X_test_s  = scaler.transform(X_test)   # ¡Solo transform, nunca fit!

# ── 4. Entrenar el modelo ────────────────────────────────
modelo = LinearRegression()
modelo.fit(X_train_s, y_train)

# ── 5. Predicciones y métricas ──────────────────────────
y_pred = modelo.predict(X_test_s)
rmse   = np.sqrt(mean_squared_error(y_test, y_pred))
r2     = r2_score(y_test, y_pred)

print("── Resultados del Modelo ─────────────────────")
print(f"R²   = {r2:.4f}  (explica el {r2*100:.1f}% de la varianza)")
print(f"RMSE = ${rmse:,.0f}")
print("\nCoeficientes del modelo:")
for feat, coef in zip(X.columns, modelo.coef_):
    print(f"  {feat:12s}: {coef:+,.0f}")
print(f"  {'intercepto':12s}: {modelo.intercept_:+,.0f}")

# ── 6. Gráfico real vs predicho ──────────────────────────
plt.figure(figsize=(7, 6))
plt.scatter(y_test, y_pred, alpha=0.6, s=30)
lims = [y.min(), y.max()]
plt.plot(lims, lims, "r--", label="Predicción perfecta")
plt.xlabel("Valores Reales")
plt.ylabel("Valores Predichos")
plt.title(f"Real vs Predicho (R² = {r2:.3f})")
plt.legend()
plt.tight_layout()
plt.show()

# ── 7. Predecir una nueva venta ──────────────────────────
nueva_venta = pd.DataFrame({
    "cantidad"  : [5],
    "precio"    : [900_000],
    "descuento" : [10]
})
nueva_scaled = scaler.transform(nueva_venta)
prediccion   = modelo.predict(nueva_scaled)[0]
print(f"\nVenta predicha: ${prediccion:,.0f}")
▶ output ── Resultados del Modelo ───────────────────── R² = 0.9812 (explica el 98.1% de la varianza) RMSE = $308,124 Coeficientes del modelo: cantidad : +1,824,512 precio : +953,871 descuento : -312,045 intercepto : +3,987,500 Venta predicha: $4,050,000
🔗 La conexión con Álgebra Lineal: Los coeficientes del modelo se calculan con β = (XᵀX)⁻¹Xᵀy. La matriz X son tus datos, la inversa viene de numpy.linalg.inv. scikit-learn hace exactamente esto (con descomposición SVD para más estabilidad numérica).
